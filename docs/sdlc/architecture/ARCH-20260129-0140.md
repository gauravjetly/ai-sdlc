## 1. Executive Summary

This architecture defines a **pipeline-agnostic, tool-flexible DevOps automation platform** that enables organizations to write workflows once and run them on any supported pipeline engine. The platform is built on the principles of **abstraction, adaptability, and composability**, allowing users to choose their preferred tools for infrastructure provisioning, container building, and Kubernetes deployment.

### Key Architecture Decisions (Revised)
- **Workflow Abstraction Layer**: Platform-native DSL that transpiles to multiple pipeline engines
- **Adapter Pattern**: Pluggable adapters for all major DevOps tools
- **Pipeline Engine Support**: Argo Workflows (primary), GitHub Actions (secondary)
- **GitOps-First Deployment**: ArgoCD and Flux as primary Kubernetes deployment methods
- **Multi-IaC Support**: Terraform, Pulumi, Crossplane, and CloudFormation via adapters
- **Standard Interface Contracts**: Versioned schemas defining inputs/outputs for each stage type

### Architecture Style
**Adapter-Based Platform** with **Workflow Abstraction** and **GitOps Deployment**

### Core Philosophy: Write Once, Run Anywhere

## 3. Workflow Abstraction Layer

### 3.1 Platform Workflow DSL

The platform defines its own workflow language that abstracts away pipeline engine specifics. Users write workflows once, and the transpiler generates engine-specific definitions.

```yaml
# Example: Platform-native workflow definition
# File: workflows/deploy-microservice.yaml
apiVersion: platform.devops/v1
kind: Workflow
metadata:
  name: deploy-microservice
  version: "1.0.0"
  description: "Complete microservice deployment workflow"

spec:
  # Input parameters with validation
  inputs:
    - name: service_name
      type: string
      required: true
      validation:
        pattern: "^[a-z][a-z0-9-]*$"
    - name: environment
      type: enum
      values: [dev, test, uat, prod]
      required: true
    - name: iac_tool
      type: enum
      values: [terraform, pulumi, crossplane, cloudformation]
      default: terraform
    - name: k8s_deploy_method
      type: enum
      values: [argocd, flux, helm, kubectl]
      default: argocd
    - name: container_builder
      type: enum
      values: [docker, kaniko, buildpacks]
      default: docker

  # Workflow stages
  stages:
    # Stage 1: Infrastructure Provisioning
    - name: provision-infrastructure
      type: infrastructure
      adapter: "{{ inputs.iac_tool }}"
      condition: "{{ inputs.environment == 'dev' || needs_infra_update }}"
      inputs:
        provider: aws
        resources:
          - type: vpc
            config:
              cidr: "10.{{ env_index }}.0.0/16"
              azs: 3
          - type: eks
            config:
              version: "1.29"
              node_count: "{{ inputs.environment == 'prod' ? 5 : 2 }}"
          - type: rds
            config:
              engine: postgresql
              version: "15"
              instance_class: "{{ inputs.environment == 'prod' ? 'db.r6i.xlarge' : 'db.t3.medium' }}"
      outputs:
        - name: vpc_id
        - name: eks_endpoint
        - name: rds_endpoint

    # Stage 2: Build Container Image
    - name: build-container
      type: container-build
      adapter: "{{ inputs.container_builder }}"
      inputs:
        source: "${git.ref}"
        dockerfile: "./Dockerfile"
        context: "."
        registry: "${aws.ecr.registry}"
        image_name: "${inputs.service_name}"
        tags:
          - "${git.sha}"
          - "${git.branch}"
          - "latest"
        build_args:
          NODE_ENV: "{{ inputs.environment }}"
      outputs:
        - name: image_uri
        - name: image_digest

    # Stage 3: Security Scanning
    - name: security-scan
      type: security
      parallel: true
      steps:
        - name: container-scan
          adapter: trivy
          inputs:
            image: "${stages.build-container.outputs.image_uri}"
            severity: "CRITICAL,HIGH"
            fail_on_vuln: true
        - name: iac-scan
          adapter: checkov
          inputs:
            path: "./infrastructure"
            framework: "{{ inputs.iac_tool }}"

    # Stage 4: Deploy to Kubernetes (GitOps)
    - name: deploy-kubernetes
      type: kubernetes-deploy
      adapter: "{{ inputs.k8s_deploy_method }}"
      method: gitops
      inputs:
        cluster: "${stages.provision-infrastructure.outputs.eks_endpoint}"
        namespace: "{{ inputs.service_name }}-{{ inputs.environment }}"
        manifests:
          source: git
          repo: "https://github.com/org/k8s-configs"
          path: "services/{{ inputs.service_name }}/{{ inputs.environment }}"
        image: "${stages.build-container.outputs.image_uri}"
        strategy: "{{ inputs.environment == 'prod' ? 'canary' : 'rolling' }}"
        canary:
          steps: [10, 25, 50, 100]
          interval: 5m
          analysis:
            metrics:
              - error_rate < 1%
              - latency_p99 < 500ms
      outputs:
        - name: deployment_status
        - name: service_urls
        - name: revision

    # Stage 5: Post-deployment verification
    - name: verify-deployment
      type: verification
      inputs:
        endpoints: "${stages.deploy-kubernetes.outputs.service_urls}"
        checks:
          - type: health
            path: /health
            expected_status: 200
          - type: smoke
            tests: "./tests/smoke"
          - type: integration
            tests: "./tests/integration"
            condition: "{{ inputs.environment != 'prod' }}"

  # Notification configuration
  notifications:
    on_success:
      - type: slack
        channel: "#deployments"
    on_failure:
      - type: pagerduty
        condition: "{{ inputs.environment == 'prod' }}"
      - type: slack
        channel: "#deployments-alerts"

  # Rollback configuration
  rollback:
    automatic: true
    conditions:
      - error_rate > 5%
      - health_check_failed
    strategy: previous_revision
```

### 3.2 Standard Interface Contracts

Each stage type has a versioned interface contract that all adapters must implement.

```yaml
# File: interfaces/infrastructure-stage-v1.yaml
apiVersion: platform.devops/v1
kind: StageInterface
metadata:
  name: infrastructure
  version: "1.0.0"

spec:
  description: "Infrastructure provisioning stage interface"

  # Supported adapters
  adapters:
    - name: terraform
      version: ">=1.5.0"
    - name: pulumi
      version: ">=3.0.0"
    - name: crossplane
      version: ">=1.0.0"
    - name: cloudformation
      version: "latest"

  # Input schema (JSON Schema format)
  input_schema:
    type: object
    required: [provider, resources]
    properties:
      provider:
        type: string
        enum: [aws, azure, gcp]
      resources:
        type: array
        items:
          type: object
          required: [type, config]
          properties:
            type:
              type: string
              enum: [vpc, eks, ecs, rds, s3, lambda, dynamodb]
            config:
              type: object
      state_backend:
        type: object
        properties:
          type:
            type: string
            enum: [s3, gcs, azureblob]
          bucket:
            type: string
          lock_table:
            type: string

  # Output schema
  output_schema:
    type: object
    properties:
      resource_ids:
        type: object
        additionalProperties:
          type: string
      endpoints:
        type: object
        additionalProperties:
          type: string
      metadata:
        type: object

  # Lifecycle hooks
  hooks:
    pre_provision:
      - validate_credentials
      - check_quotas
    post_provision:
      - tag_resources
      - update_inventory
    pre_destroy:
      - backup_state
      - notify_stakeholders
```

```yaml

          source:
            type: string

            type: string

            type: string

            type: string

            type: object
      image:
        type: string
        description: "Container image to deploy"

        type: string
        enum: [rolling, blue-green, canary, recreate]
        default: rolling

        type: object
        properties:
          steps:
            type: array
            items:
              type: integer
          interval:
            type: string
          analysis:
            type: object

  output_schema:
    type: object
    properties:
      deployment_status:
        type: string
        enum: [success, failed, in_progress, rolled_back]
      service_urls:
        type: array
        items:
          type: string
      revision:
        type: string
      rollback_revision:
        type: string
```

```yaml
on:

    outputs:
      eks_endpoint: ${{ steps.provision.outputs.eks_endpoint }}

      - uses: actions/checkout@v4
      - name: Run Infrastructure Adapter
        id: provision
        uses: platform/infrastructure-adapter-action@v1

            - type: vpc
            - type: eks
            - type: rds

    runs-on: ubuntu-latest
    outputs:
      image_uri: ${{ steps.build.outputs.image_uri }}
    steps:
      - uses: actions/checkout@v4
      - name: Run Container Build Adapter
        id: build
        uses: platform/container-build-adapter-action@v1
        with:
          adapter: docker
          image_name: ${{ inputs.service_name }}

    needs: [build-container]
    runs-on: ubuntu-latest

      - name: Run Security Scan

        with:
          image: ${{ needs.build-container.outputs.image_uri }}

    needs: [provision-infrastructure, security-scan]
    runs-on: ubuntu-latest
    steps:
      - name: Run K8s Deploy Adapter
        uses: platform/k8s-deploy-adapter-action@v1
        with:
          adapter: ${{ inputs.k8s_deploy_method }}
          cluster: ${{ needs.provision-infrastructure.outputs.eks_endpoint }}
```

**ArgoCD Adapter (GitOps):**

```typescript
// File: adapters/kubernetes/argocd/adapter.ts

import { KubernetesDeployAdapter, KubernetesDeployInput, KubernetesDeployOutput } from '../interface';
import { ArgoClient } from './client';

export class ArgoCDAdapter implements KubernetesDeployAdapter {
  readonly name = 'argocd';
  readonly version = '2.9.0';
  readonly method = 'gitops' as const;

  private client: ArgoClient;

  async initialize(config: AdapterConfig): Promise<void> {
    this.client = new ArgoClient({
      server: config.argoServer,
      token: config.argoToken
    });
  }

  async deploy(input: KubernetesDeployInput): Promise<KubernetesDeployOutput> {
    // Create or update ArgoCD Application
    const app = await this.client.createOrUpdateApplication({
      metadata: {
        name: `${input.namespace}-app`,
        namespace: 'argocd'
      },
      spec: {
        project: 'default',
        source: {
          repoURL: input.manifests.repo,
          targetRevision: 'HEAD',
          path: input.manifests.path,
          kustomize: input.image ? {
            images: [input.image]
          } : undefined
        },
        destination: {
          server: input.cluster,
          namespace: input.namespace
        },
        syncPolicy: {
          automated: {
            prune: true,
            selfHeal: true
          },
          syncOptions: ['CreateNamespace=true']
        }
      }
    });

    // Trigger sync
    await this.sync();

    // Wait for deployment
    const ready = await this.waitForReady(300);

    return {
      status: ready ? 'success' : 'failed',
      serviceUrls: await this.getServiceUrls(input.namespace),
      revision: app.status.sync.revision
    };
  }

  async sync(): Promise<SyncResult> {
    return this.client.sync();
  }

  async refresh(): Promise<void> {
    return this.client.refresh();
  }

  async rollback(revision: string): Promise<KubernetesDeployOutput> {
    await this.client.rollback(revision);
    return this.getStatus();
  }

  async waitForReady(timeout: number): Promise<boolean> {
    const deadline = Date.now() + timeout * 1000;
    while (Date.now() < deadline) {
      const status = await this.client.getAppStatus();
      if (status.health === 'Healthy' && status.sync === 'Synced') {
        return true;
      }
      await new Promise(resolve => setTimeout(resolve, 5000));
    }
    return false;
  }
}
```

**Flux Adapter (GitOps):**

```typescript
// File: adapters/kubernetes/flux/adapter.ts

import { KubernetesDeployAdapter, KubernetesDeployInput, KubernetesDeployOutput } from '../interface';
import { KubernetesClient } from '../k8s-client';

export class FluxAdapter implements KubernetesDeployAdapter {
  readonly name = 'flux';
  readonly version = '2.2.0';
  readonly method = 'gitops' as const;

  private k8s: KubernetesClient;

  async initialize(config: AdapterConfig): Promise<void> {
    this.k8s = new KubernetesClient(config.kubeconfig);
  }

  async deploy(input: KubernetesDeployInput): Promise<KubernetesDeployOutput> {
    // Create GitRepository source
    await this.k8s.apply({
      apiVersion: 'source.toolkit.fluxcd.io/v1',
      kind: 'GitRepository',
      metadata: {
        name: `${input.namespace}-repo`,
        namespace: 'flux-system'
      },
      spec: {
        interval: '1m',
        url: input.manifests.repo,
        ref: { branch: 'main' }
      }
    });

    // Create Kustomization
    await this.k8s.apply({
      apiVersion: 'kustomize.toolkit.fluxcd.io/v1',
      kind: 'Kustomization',
      metadata: {
        name: `${input.namespace}-kustomization`,
        namespace: 'flux-system'
      },
      spec: {
        interval: '5m',
        path: input.manifests.path,
        prune: true,
        sourceRef: {
          kind: 'GitRepository',
          name: `${input.namespace}-repo`
        },
        targetNamespace: input.namespace,
        images: input.image ? [{
          name: '*',
          newName: input.image.split(':')[0],
          newTag: input.image.split(':')[1]
        }] : undefined
      }
    });

    // Trigger reconciliation
    await this.sync();

    const ready = await this.waitForReady(300);

    return {
      status: ready ? 'success' : 'failed',
      serviceUrls: await this.getServiceUrls(input.namespace),
      revision: await this.getCurrentRevision(input.namespace)
    };
  }

  async sync(): Promise<SyncResult> {
    // Annotate to trigger immediate reconciliation
    await this.k8s.annotate(
      'kustomization',
      'reconcile.fluxcd.io/requestedAt',
      new Date().toISOString()
    );
    return { status: 'syncing' };
  }
}
```
