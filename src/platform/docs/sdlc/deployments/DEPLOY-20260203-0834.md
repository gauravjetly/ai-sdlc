## 6. Usage Guide

### 6.1 Basic Usage

```typescript
import { generateTerraform } from './services/terraform-executor/generator';

const result = await generateTerraform({
  nodes: [
    {
      id: 'vpc-1',
      type: 'vpc',
      data: {
        id: 'vpc-1',
        name: 'my-vpc',
        serviceType: 'vpc',
        category: 'networking',
        cidrBlock: '10.0.0.0/16',
        enableDnsHostnames: true,
        enableDnsSupport: true,
        tags: {},
        status: 'configured',
        createdAt: new Date().toISOString(),
        updatedAt: new Date().toISOString(),
      },
      position: { x: 0, y: 0 },
    },
  ],
  edges: [],
  environment: 'dev',
  projectName: 'my-project',
  region: 'us-east-1',
  commonTags: { Team: 'Platform' },
  outputPath: './terraform-output',
});

if (result.success) {
  console.log('Generated modules:', result.modules.map(m => m.name));
}
```

### 6.2 Preview Single Node

```typescript
import { previewNode } from './services/terraform-executor/generator';

const preview = previewNode({
  id: 's3-1',
  type: 's3-bucket',
  data: {
    id: 's3-1',
    name: 'my-bucket',
    serviceType: 's3-bucket',
    category: 'storage',
    bucketName: 'my-unique-bucket-name',
    versioning: true,
    encryption: { enabled: true, algorithm: 'AES256' },
    blockPublicAccess: {
      blockPublicAcls: true,
      ignorePublicAcls: true,
      blockPublicPolicy: true,
      restrictPublicBuckets: true,
    },
    tags: {},
    status: 'configured',
    createdAt: new Date().toISOString(),
    updatedAt: new Date().toISOString(),
  },
  position: { x: 0, y: 0 },
});

console.log(preview.mainTf);
```
