```typescript
describe('useDevOpsValidation', () => {
  describe('validateCICDConfig', () => {
    it('should return error when repository URL is empty', () => {
      const config = { ...DEFAULT_CICD_CONFIG, sourceControl: { ...DEFAULT_CICD_CONFIG.sourceControl, repositoryUrl: '' }};
      const errors = validateCICDConfig(config);
      expect(errors).toContainEqual(expect.objectContaining({ code: 'CICD_NO_REPO' }));
    });

    it('should return error for invalid git URL format', () => {
      const config = { ...DEFAULT_CICD_CONFIG, sourceControl: { ...DEFAULT_CICD_CONFIG.sourceControl, repositoryUrl: 'not-a-url' }};
      const errors = validateCICDConfig(config);
      expect(errors).toContainEqual(expect.objectContaining({ code: 'CICD_INVALID_REPO' }));
    });

    it('should pass for valid GitHub URL', () => {
      const config = { ...DEFAULT_CICD_CONFIG, sourceControl: { ...DEFAULT_CICD_CONFIG.sourceControl, repositoryUrl: 'https://github.com/org/repo' }};
      const errors = validateCICDConfig(config);
      expect(errors.filter(e => e.code === 'CICD_NO_REPO')).toHaveLength(0);
    });

    it('should warn when no test stage configured', () => {
      const config = { ...DEFAULT_CICD_CONFIG, buildPipeline: { ...DEFAULT_CICD_CONFIG.buildPipeline, stages: [] }};
      const errors = validateCICDConfig(config);
      expect(errors).toContainEqual(expect.objectContaining({ code: 'CICD_NO_STAGES', severity: 'error' }));
    });
  });
});
```
