## 1. Executive Summary

This document defines the architecture for implementing pre-built infrastructure templates and database seeding functionality for the Deltek Catalyst Infrastructure Designer. The solution provides 7 production-ready templates covering common AWS architecture patterns, with a modular seed script architecture for maintainability.

### 1.1 Key Architectural Decisions
- Modular template architecture (one file per template)
- Shared utility functions for node/edge creation
- ReactFlow-compatible data structures
- Prisma upsert for idempotent seeding
- Type-safe template definitions with full TypeScript support

### 3.1 Core Template Interface

```typescript
// prisma/seeds/templates/types.ts

import { TemplateCategory, TemplateVisibility, LayerType } from '@prisma/client';
import type { Node, Edge } from 'reactflow';

export interface TemplateDefinition {
  name: string;
  description: string;
  category: TemplateCategory;
  visibility: TemplateVisibility;
  layerType: LayerType;
  version: string;
  tags: string[];
  templateData: {
    nodes: TemplateNode[];
    edges: TemplateEdge[];
    metadata: TemplateMetadata;
  };
}

export interface TemplateNode {
  id: string;
  type: string;           // e.g., 'securityGroup', 'loadBalancer', 'ec2Instance'
  position: { x: number; y: number };
  data: Record<string, unknown>;
}

export interface TemplateEdge {
  id: string;
  source: string;
  target: string;
  sourceHandle?: string;
  targetHandle?: string;
  type?: string;
  animated?: boolean;
  label?: string;
}

export interface TemplateMetadata {
  author: string;
  version: string;
  tested: boolean;
  compliance: string[];
  estimatedCost: {
    dev: number;
    staging?: number;
    prod: number;
    currency: string;
  };
  prerequisites: string[];
  customizationGuide: string;
  deploymentTime: string;
  components: ComponentSummary[];
}

export interface ComponentSummary {
  type: string;
  count: number;
  description: string;
}
```

### 5.2 Edge Factory

```typescript
// prisma/seeds/templates/utils/edge-factory.ts

import { v4 as uuid } from 'uuid';
import type { TemplateEdge } from '../types';

export function createEdge(
  source: string,
  target: string,
  options?: {
    sourceHandle?: string;
    targetHandle?: string;
    label?: string;
    animated?: boolean;
  }
): TemplateEdge {
  return {
    id: uuid(),
    source,
    target,
    sourceHandle: options?.sourceHandle,
    targetHandle: options?.targetHandle,
    type: 'default',
    animated: options?.animated ?? false,
    label: options?.label,
  };
}

// Connection type helpers
export function createSecurityGroupAttachment(sgId: string, resourceId: string): TemplateEdge {
  return createEdge(sgId, resourceId, { sourceHandle: 'security-attachment' });
}

export function createLoadBalancerTarget(albId: string, targetId: string): TemplateEdge {
  return createEdge(albId, targetId, { sourceHandle: 'target', label: 'target' });
}

export function createSubnetPlacement(subnetId: string, resourceId: string): TemplateEdge {
  return createEdge(subnetId, resourceId, { sourceHandle: 'subnet-placement' });
```

### 7.1 Main Seed Orchestrator

```typescript
// prisma/seeds/index.ts

import { PrismaClient } from '@prisma/client';
import { seedTemplates } from './templates.seed';

const prisma = new PrismaClient();

async function main() {
  console.log('Starting database seeding...');

  try {
    // Seed templates
    await seedTemplates(prisma);

    console.log('Database seeding completed successfully!');
  } catch (error) {
    console.error('Error during seeding:', error);
    throw error;
  }
}

main()
  .catch((e) => {
    console.error(e);
    process.exit(1);
  })
  .finally(async () => {
    await prisma.$disconnect();
  });
```

### 7.2 Template Seeding Logic

```typescript
// prisma/seeds/templates.seed.ts

import { PrismaClient } from '@prisma/client';
import { TEMPLATES } from './templates';
import type { TemplateDefinition } from './templates/types';

export async function seedTemplates(prisma: PrismaClient): Promise<void> {
  console.log(`Seeding ${TEMPLATES.length} templates...`);

  for (const template of TEMPLATES) {
    await upsertTemplate(prisma, template);
  }

  console.log('Template seeding complete!');
}

async function upsertTemplate(
  prisma: PrismaClient,
  template: TemplateDefinition
): Promise<void> {
  const { name, description, category, visibility, layerType, version, tags, templateData } = template;

  // Validate template before inserting
  validateTemplate(template);

  const result = await prisma.designTemplate.upsert({
    where: { name },
    update: {
      description,
      category,
      visibility,
      layerType,
      version,
      tags,
      templateData: templateData as any,
      updatedAt: new Date(),
    },
    create: {
      name,
      description,
      category,
      visibility,
      layerType,
      version,
      tags,
      templateData: templateData as any,
      usageCount: 0,
      createdBy: 'system',
    },
  });

  console.log(`  [+] ${result.name} (${result.id})`);
}

function validateTemplate(template: TemplateDefinition): void {
  const { name, templateData } = template;
  const { nodes, edges, metadata } = templateData;

  // Validate nodes
  if (!nodes || nodes.length === 0) {
    throw new Error(`Template "${name}" has no nodes`);
  }

  // Validate edges reference valid nodes
  const nodeIds = new Set(nodes.map(n => n.id));
  for (const edge of edges) {
    if (!nodeIds.has(edge.source)) {
      throw new Error(`Template "${name}": Edge ${edge.id} has invalid source ${edge.source}`);
    }
    if (!nodeIds.has(edge.target)) {
      throw new Error(`Template "${name}": Edge ${edge.id} has invalid target ${edge.target}`);
    }
  }

  // Validate metadata
  if (!metadata.estimatedCost) {
    throw new Error(`Template "${name}" missing cost estimate`);
  }
}
```

## 8. ADR: Template Architecture Decisions

### ADR-031: Modular Template File Structure
**Status**: Accepted

**Context**: Need to organize 7+ templates in a maintainable way.

**Decision**: One TypeScript file per template with shared utilities.

**Rationale**:
- Easy to add/modify individual templates
- Shared utilities reduce code duplication
- Clear separation of concerns
- Easy to test individual templates

### ADR-032: ReactFlow-Compatible Data Structure
**Status**: Accepted

**Context**: Templates must work with the Visual Designer.

**Decision**: Use ReactFlow Node and Edge interfaces directly.

**Rationale**:
- No transformation needed at runtime
- Type safety with existing interfaces
- Consistent with frontend expectations

### ADR-033: Upsert for Idempotent Seeding
**Status**: Accepted

**Context**: Seed script may run multiple times during development.

**Decision**: Use Prisma upsert with name as unique key.

**Rationale**:
- Safe to run repeatedly
- Updates existing templates
- Prevents duplicate entries
