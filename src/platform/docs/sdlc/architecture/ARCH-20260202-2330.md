## 1. Executive Summary

This architecture document defines the component hierarchy, type system, and integration patterns for 20 enhanced AWS node components in the Deltek Catalyst Visual Designer. The design follows a plugin-based architecture enabling easy extension while maintaining type safety and performance.

### 3.1 Base Types

```typescript
// types/base.types.ts

/**
 * Node categories for color-coding and organization
 */
export type NodeCategory =
  | 'security'
  | 'networking'
  | 'compute'
  | 'storage'
  | 'monitoring';

/**
 * AWS service identifiers for each node type
 */
export type AWSServiceType =
  // Security
  | 'security-group'
  | 'iam-role'
  | 'iam-policy'
  | 'kms-key'
  // Networking
  | 'load-balancer'
  | 'route53'
  | 'cloudfront'
  | 'vpn-gateway'
  | 'transit-gateway'
  // Compute
  | 'ec2-instance'
  | 'auto-scaling-group'
  | 'ecs-cluster'
  | 'lambda-function'
  // Storage
  | 's3-bucket'
  | 'dynamodb-table'
  | 'elasticache-cluster'
  | 'efs-filesystem'
  // Monitoring
  | 'cloudwatch-alarm'
  | 'sns-topic'
  | 'sqs-queue';

/**
 * Connection handle types for typed connections
 */
export type HandleType =
  | 'vpc-attachment'       // Can attach to VPC
  | 'subnet-placement'     // Can be placed in subnet
  | 'security-attachment'  // Can attach security groups
  | 'iam-role'            // Can assume IAM role
  | 'encryption'          // Can use KMS encryption
  | 'target'              // Can be a target (LB, ASG)
  | 'trigger'             // Can trigger (events)
  | 'notification'        // Can send/receive notifications
  | 'dns-alias'           // Can be DNS aliased
  | 'origin';             // Can be origin (CloudFront)

/**
 * Node configuration status
 */
export type NodeStatus =
  | 'unconfigured'  // Missing required fields
  | 'configured'    // All required fields valid
  | 'warning'       // Has validation warnings
  | 'error'         // Has validation errors
  | 'deployed';     // Successfully deployed

/**
 * Validation severity levels
 */
export type ValidationSeverity = 'error' | 'warning' | 'info';

/**
 * Validation result for a node
 */
export interface ValidationResult {
  valid: boolean;
  errors: ValidationMessage[];
  warnings: ValidationMessage[];
}

export interface ValidationMessage {
  code: string;
  message: string;
  field?: string;
  severity: ValidationSeverity;
}

/**
 * Connection point definition
 */
export interface HandleDefinition {
  id: string;
  type: HandleType;
  position: 'top' | 'right' | 'bottom' | 'left';
  label?: string;
  maxConnections?: number; // undefined = unlimited
}

/**
 * Base interface for all node data
 */
export interface BaseNodeData {
  // Identity
  id: string;
  name: string;
  serviceType: AWSServiceType;
  category: NodeCategory;

  // Configuration
  tags: Record<string, string>;

  // Status
  status: NodeStatus;
  validationResult?: ValidationResult;

  // Cost
  estimatedMonthlyCost?: number;

  // Metadata
  createdAt: string;
  updatedAt: string;
}

/**
 * Node metadata for registry
 */
export interface NodeMetadata {
  serviceType: AWSServiceType;
  category: NodeCategory;
  displayName: string;
  description: string;
  icon: string;
  color: string;
  defaultWidth: number;
  defaultHeight: number;
  handles: {
    inputs: HandleDefinition[];
    outputs: HandleDefinition[];
  };
  defaultData: Partial<BaseNodeData>;
  terraformResource: string;
  awsDocUrl: string;
}

/**
 * Connection rule definition
 */
export interface ConnectionRule {
  sourceType: AWSServiceType;
  targetType: AWSServiceType;
  sourceHandle: HandleType;
  targetHandle: HandleType;
  connectionType: string;
  description: string;
  maxConnections?: number;
  validator?: (source: BaseNodeData, target: BaseNodeData) => ValidationResult;
}
```

### 4.2 Node Registry

```typescript
// NodeRegistry.ts

import { AWSServiceType, NodeMetadata, NodeCategory } from './types';

// Import all node metadata
import { securityGroupMetadata } from './security/SecurityGroupNode';
import { iamRoleMetadata } from './security/IAMRoleNode';
// ... more imports

/**
 * Central registry of all node types
 */
class NodeRegistryClass {
  private nodes: Map<AWSServiceType, NodeMetadata> = new Map();

  register(metadata: NodeMetadata): void {
    this.nodes.set(metadata.serviceType, metadata);
  }

  get(serviceType: AWSServiceType): NodeMetadata | undefined {
    return this.nodes.get(serviceType);
  }

  getByCategory(category: NodeCategory): NodeMetadata[] {
    return Array.from(this.nodes.values())
      .filter(node => node.category === category);
  }

  getAll(): NodeMetadata[] {
    return Array.from(this.nodes.values());
  }

  getReactFlowNodeTypes(): Record<string, React.ComponentType<NodeProps>> {
    const types: Record<string, React.ComponentType<NodeProps>> = {};
    for (const [serviceType, metadata] of this.nodes) {
      types[serviceType] = metadata.component;
    }
    return types;
  }
}

export const NodeRegistry = new NodeRegistryClass();
```

```typescript
// validation/ConnectionValidator.ts

import { Connection, Edge } from 'reactflow';
import {
  AWSServiceType,
  ConnectionRule,
  ValidationResult,
  BaseNodeData
} from '../types';

/**
 * Connection rules matrix
 */
const CONNECTION_RULES: ConnectionRule[] = [
  // Security Group connections
  {
    sourceType: 'security-group',
    targetType: 'ec2-instance',
    sourceHandle: 'security-attachment',
    targetHandle: 'security-attachment',
    connectionType: 'security-attachment',
    description: 'Attach security group to EC2 instance',
  },
  {
    sourceType: 'security-group',
    targetType: 'load-balancer',
    sourceHandle: 'security-attachment',
    targetHandle: 'security-attachment',
    connectionType: 'security-attachment',
    description: 'Attach security group to load balancer',
  },
  // ... more rules

  // IAM Role connections
  {
    sourceType: 'iam-role',
    targetType: 'ec2-instance',
    sourceHandle: 'iam-role',
    targetHandle: 'iam-role',
    connectionType: 'instance-profile',
    description: 'Assign IAM role to EC2 via instance profile',
  },
  {
    sourceType: 'iam-role',
    targetType: 'lambda-function',
    sourceHandle: 'iam-role',
    targetHandle: 'iam-role',
    connectionType: 'execution-role',
    description: 'Assign execution role to Lambda function',
  },
  // ... more rules

  // KMS Key connections
  {
    sourceType: 'kms-key',
    targetType: 's3-bucket',
    sourceHandle: 'encryption',
    targetHandle: 'encryption',
    connectionType: 'encryption',
    description: 'Encrypt S3 bucket with KMS key',
  },
  // ... more rules

  // Load Balancer connections
  {
    sourceType: 'load-balancer',
    targetType: 'ec2-instance',
    sourceHandle: 'target',
    targetHandle: 'target',
    connectionType: 'target-registration',
    description: 'Register EC2 as load balancer target',
  },
  // ... more rules
];

/**
 * Validates connections between nodes
 */
export class ConnectionValidator {
  private rules: ConnectionRule[];

  constructor() {
    this.rules = CONNECTION_RULES;
  }

  /**
   * Check if a connection is valid before it's created
   */
  isValidConnection(
    sourceNode: BaseNodeData,
    targetNode: BaseNodeData,
    sourceHandle?: string,
    targetHandle?: string
  ): ValidationResult {
    // Find matching rule
    const rule = this.rules.find(r =>
      r.sourceType === sourceNode.serviceType &&
      r.targetType === targetNode.serviceType &&
      (sourceHandle === undefined || r.sourceHandle === sourceHandle) &&
      (targetHandle === undefined || r.targetHandle === targetHandle)
    );

    if (!rule) {
      return {
        valid: false,
        errors: [{
          code: 'INVALID_CONNECTION',
          message: `Cannot connect ${sourceNode.serviceType} to ${targetNode.serviceType}`,
          severity: 'error',
        }],
        warnings: [],
      };
    }

    // Run custom validator if defined
    if (rule.validator) {
      return rule.validator(sourceNode, targetNode);
    }

    return { valid: true, errors: [], warnings: [] };
  }

  /**
   * Get all valid target types for a source node
   */
  getValidTargets(sourceType: AWSServiceType): AWSServiceType[] {
    return this.rules
      .filter(r => r.sourceType === sourceType)
      .map(r => r.targetType);
  }

  /**
   * Get all valid source types for a target node
   */
  getValidSources(targetType: AWSServiceType): AWSServiceType[] {
    return this.rules
      .filter(r => r.targetType === targetType)
      .map(r => r.sourceType);
  }
}

export const connectionValidator = new ConnectionValidator();
```

```typescript
// terraform/TerraformGenerator.ts

import { DesignNode, DesignEdge } from '../../../contexts/DesignWizardContext';
import { BaseNodeData, AWSServiceType } from '../types';

export interface TerraformModule {
  name: string;
  main: string;

  outputs: string;
}

export interface TerraformGeneratorContext {
  nodes: DesignNode[];
  edges: DesignEdge[];
  environment: 'dev' | 'staging' | 'prod';
  region: string;
  projectName: string;
}

/**
 * Abstract base class for node-specific generators
 */
export abstract class NodeTerraformGenerator<T extends BaseNodeData> {
  abstract serviceType: AWSServiceType;

  abstract generateResource(node: T, context: TerraformGeneratorContext): string;
  abstract generateVariables(node: T): string;
  abstract generateOutputs(node: T): string;

  protected generateTags(tags: Record<string, string>, name: string): string {
    const allTags = {
      Name: name,
      Environment: '${var.environment}',
      ManagedBy: 'terraform',
      ...tags,
    };

    return `
  tags = {
${Object.entries(allTags).map(([k, v]) => `    "${k}" = "${v}"`).join('\n')}
  }`;
  }
}

/**
 * Main Terraform generator
 */
export class TerraformGenerator {
  private generators: Map<AWSServiceType, NodeTerraformGenerator<any>> = new Map();

  register<T extends BaseNodeData>(generator: NodeTerraformGenerator<T>): void {
    this.generators.set(generator.serviceType, generator);
  }

  generate(context: TerraformGeneratorContext): TerraformModule[] {
    const modules: TerraformModule[] = [];

    // Group nodes by category
    const nodesByCategory = this.groupByCategory(context.nodes);

    // Generate module for each category
    for (const [category, nodes] of Object.entries(nodesByCategory)) {
      const resources: string[] = [];
      const variables: string[] = [];
      const outputs: string[] = [];

      for (const node of nodes) {
        const generator = this.generators.get(node.data.serviceType);
        if (generator) {
          resources.push(generator.generateResource(node.data, context));
          variables.push(generator.generateVariables(node.data));
          outputs.push(generator.generateOutputs(node.data));
        }
      }

      modules.push({
        name: category,
        main: this.wrapWithHeader(resources.join('\n\n')),
        variables: this.wrapVariables(variables.join('\n\n')),
        outputs: this.wrapOutputs(outputs.join('\n\n')),
      });
    }

    return modules;
  }

  private groupByCategory(nodes: DesignNode[]): Record<string, DesignNode[]> {
    return nodes.reduce((acc, node) => {
      const category = node.data.category;
      if (!acc[category]) acc[category] = [];
      acc[category].push(node);
      return acc;
    }, {} as Record<string, DesignNode[]>);
  }

  private wrapWithHeader(content: string): string {
    return `# Generated by Deltek Catalyst Visual Designer
# Do not edit manually - changes will be overwritten

terraform {
  required_version = ">= 1.5.0"
  required_providers {
    aws = {
      source  = "hashicorp/aws"
      version = "~> 5.0"
    }
  }
}

${content}`;
  }

  private wrapVariables(content: string): string {
    return `# Variables for Terraform module
# Generated by Deltek Catalyst Visual Designer

variable "environment" {
  description = "Deployment environment"
  type        = string
}

variable "region" {
  description = "AWS region"
  type        = string
}

${content}`;
  }

  private wrapOutputs(content: string): string {
    return `# Outputs for Terraform module
# Generated by Deltek Catalyst Visual Designer

${content}`;
  }
}
```

```typescript
// terraform/generators/securityGenerator.ts

import { NodeTerraformGenerator, TerraformGeneratorContext } from '../TerraformGenerator';
import { SecurityGroupNodeData, SecurityRule } from '../../types/security.types';

export class SecurityGroupTerraformGenerator extends NodeTerraformGenerator<SecurityGroupNodeData> {
  serviceType = 'security-group' as const;

  generateResource(node: SecurityGroupNodeData, context: TerraformGeneratorContext): string {
    const ingressRules = node.ingressRules.map(rule => this.generateRule('ingress', rule)).join('\n');
    const egressRules = node.egressRules.map(rule => this.generateRule('egress', rule)).join('\n');

    return `
resource "aws_security_group" "${this.sanitizeName(node.name)}" {
  name        = "${node.name}"
  description = "${node.description}"
  vpc_id      = var.vpc_id

${ingressRules}

${egressRules}

${this.generateTags(node.tags, node.name)}
}`;
  }

  generateVariables(node: SecurityGroupNodeData): string {
    return `
variable "vpc_id" {
  description = "VPC ID for security group ${node.name}"
  type        = string
}`;
  }

  generateOutputs(node: SecurityGroupNodeData): string {
    const resourceName = this.sanitizeName(node.name);
    return `
output "${resourceName}_id" {
  description = "ID of security group ${node.name}"
  value       = aws_security_group.${resourceName}.id
}

output "${resourceName}_arn" {
  description = "ARN of security group ${node.name}"
  value       = aws_security_group.${resourceName}.arn
}`;
  }

  private generateRule(type: 'ingress' | 'egress', rule: SecurityRule): string {
    const cidrBlock = rule.cidrBlocks?.length
      ? `cidr_blocks = [${rule.cidrBlocks.map(c => `"${c}"`).join(', ')}]`
      : '';

    const securityGroups = rule.securityGroupIds?.length
      ? `security_groups = [${rule.securityGroupIds.map(s => `"${s}"`).join(', ')}]`
      : '';

    return `
  ${type} {
    description = "${rule.description || ''}"
    from_port   = ${rule.fromPort}
    to_port     = ${rule.toPort}
    protocol    = "${rule.protocol}"
    ${cidrBlock}
    ${securityGroups}
  }`;
  }

  private sanitizeName(name: string): string {
    return name.toLowerCase().replace(/[^a-z0-9]/g, '_');
  }
}
```

## 6. Cost Estimation

### 6.1 Cost Estimator Architecture

```typescript
// cost/CostEstimator.ts

import { BaseNodeData, AWSServiceType } from '../types';

export interface CostEstimate {
  hourly: number;
  monthly: number;
  yearly: number;
  breakdown: CostBreakdownItem[];
  currency: 'USD';
}

export interface CostBreakdownItem {
  component: string;
  description: string;
  hourly: number;
  monthly: number;
  unit: string;
  quantity: number;
}

/**
 * Abstract base class for node-specific cost estimators
 */
export abstract class NodeCostEstimator<T extends BaseNodeData> {
  abstract serviceType: AWSServiceType;
  abstract estimate(node: T, region: string): Promise<CostEstimate>;
}

/**
 * Main cost estimation service
 */
export class CostEstimator {
  private estimators: Map<AWSServiceType, NodeCostEstimator<any>> = new Map();
  private pricingCache: PricingCache;

  constructor() {
    this.pricingCache = new PricingCache();
  }

  register<T extends BaseNodeData>(estimator: NodeCostEstimator<T>): void {
    this.estimators.set(estimator.serviceType, estimator);
  }

  async estimateNode(node: BaseNodeData, region: string): Promise<CostEstimate> {
    const estimator = this.estimators.get(node.serviceType);
    if (!estimator) {
      return this.zeroEstimate();
    }
    return estimator.estimate(node, region);
  }

  async estimateDesign(nodes: BaseNodeData[], region: string): Promise<CostEstimate> {
    const estimates = await Promise.all(
      nodes.map(node => this.estimateNode(node, region))
    );

    return this.aggregateEstimates(estimates);
  }

  private zeroEstimate(): CostEstimate {
    return {
      hourly: 0,
      monthly: 0,
      yearly: 0,
      breakdown: [],
      currency: 'USD',
    };
  }

  private aggregateEstimates(estimates: CostEstimate[]): CostEstimate {
    const hourly = estimates.reduce((sum, e) => sum + e.hourly, 0);
    const monthly = estimates.reduce((sum, e) => sum + e.monthly, 0);
    const breakdown = estimates.flatMap(e => e.breakdown);

    return {
      hourly,
      monthly,
      yearly: monthly * 12,
      breakdown,
      currency: 'USD',
    };
  }
}

/**
 * Pricing data cache with TTL
 */
class PricingCache {
  private cache: Map<string, { data: any; expiry: number }> = new Map();
  private TTL_MS = 60 * 60 * 1000; // 1 hour

  async get(key: string): Promise<any | null> {
    const entry = this.cache.get(key);
    if (!entry || Date.now() > entry.expiry) {
      return null;
    }
    return entry.data;
  }

  set(key: string, data: any): void {
    this.cache.set(key, {
      data,
      expiry: Date.now() + this.TTL_MS,
    });
  }
}
```

## 8. Performance Optimizations

### 8.1 Virtualization Strategy

```typescript
// hooks/useVirtualizedNodes.ts

import { useCallback, useMemo } from 'react';
import { useViewport, useReactFlow } from 'reactflow';
import { DesignNode } from '../contexts/DesignWizardContext';

/**
 * Hook for viewport-based node virtualization
 * Only renders nodes within the visible viewport + buffer
 */
export function useVirtualizedNodes(nodes: DesignNode[], buffer = 200) {
  const { x, y, zoom } = useViewport();
  const { getViewport } = useReactFlow();

  const visibleNodes = useMemo(() => {
    const viewport = getViewport();
    const viewportWidth = window.innerWidth / zoom;
    const viewportHeight = window.innerHeight / zoom;

    const visibleArea = {
      left: -x / zoom - buffer,
      right: -x / zoom + viewportWidth + buffer,
      top: -y / zoom - buffer,
      bottom: -y / zoom + viewportHeight + buffer,
    };

    return nodes.filter(node => {
      const { x: nodeX, y: nodeY } = node.position;
      const nodeWidth = node.width || 200;
      const nodeHeight = node.height || 100;

      return (
        nodeX + nodeWidth >= visibleArea.left &&
        nodeX <= visibleArea.right &&
        nodeY + nodeHeight >= visibleArea.top &&
        nodeY <= visibleArea.bottom
      );
    });
  }, [nodes, x, y, zoom, buffer, getViewport]);

  return visibleNodes;
}
```
